{{/* If Hugo has been able to access the image, it's loaded at .image, and we have access to additional attributes. If Hugo hasn't been able to access the image, it's loaded at .file, with only its path available. We set the Hugo variables to strings to allow for logical operations and replacements. */}} {{- $file := .file -}} {{- $image := .image -}} {{- $dir := .dir -}} {{- $alt := .alt -}} {{- $cap := .cap -}} {{- $classes := .classes -}} {{- $bundle := .bundle -}} {{- if eq $bundle true -}} {{ $file = path.Join $dir $file }} {{ if in $classes "image_svg" }} {{ if strings.HasPrefix .file "http" }} {{ $file = .file }} {{- end -}} {{- end -}} {{- end -}}
